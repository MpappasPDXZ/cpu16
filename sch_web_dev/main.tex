    % Job Schedule Report (JSR)
    \begin{tikzpicture}
        \node[fill=black!10, rounded corners=3pt, inner sep=3pt, text width=\linewidth-6pt, align=left] {\textbf{1. Job Schedule Report (JSR)}};
    \end{tikzpicture}
    
    \vspace{0.1em}
    
    {\footnotesize
    The JSR is a \textbf{schedule version comparison} tool that analyzes project schedules against baselines and prior periods, integrated with financial metadata.
    
    \vspace{0.15em}
    
    \textbf{Schedule Comparison (Core Function):}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item \textbf{Current Period} vs \textbf{Current Baseline} (primary comparison)
        \item \textbf{Prior Period} for trend analysis
        \item Generates KPIs from schedule variances
    \end{itemize}
    
    \vspace{0.15em}
    
    \textbf{Integrated Metadata:}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item \textbf{JOR:} Job Operations Report (Contract Value, Total Cost)
        \item \textbf{Cost Report:} Labor hours, earned values, forecasts
        \item \textbf{CRM:} Project/opportunity information
    \end{itemize}
    }
    
    \vspace{0.3em}
    
    % Key Performance Indicators
    \begin{tikzpicture}
        \node[fill=black!10, rounded corners=3pt, inner sep=3pt, text width=\linewidth-6pt, align=left] {\textbf{Key Performance Indicators}};
    \end{tikzpicture}
    
    \vspace{0.1em}
    
    {\footnotesize
    \textbf{Schedule Quality \& Planning:}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item \textbf{Quality Score:} Activities, Hard Constraints, Missing Logic, Invalid Dates
        \item \textbf{Planning Score:} BL/CM Remaining, T4 metrics, Compression
        \item \textbf{Float Analysis:} Schedule slack assessment
    \end{itemize}
    
    \vspace{0.15em}
    \textbf{JTD Performance (from Cost Report):}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item CR Spent vs Earned vs Plan
        \item SPI (Schedule Performance Index)
        \item \% Complete, Earned/Spent ratio
    \end{itemize}
    \textit{\scriptsize Account codes 30.06 + 50-87 (direct); 20, 30, 30.12, 88 (indirect)}
    
    \vspace{0.15em}
    \textbf{Total Performance:}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item CR CE (Current Estimate), CR FC (Forecast)
        \item CE/FC ratio, BL/CM comparison
    \end{itemize}
    }

    \vspace{0.3em}

    % Extraction Workflow
    \begin{tikzpicture}
        \node[fill=black!10, rounded corners=3pt, inner sep=3pt, text width=\linewidth-6pt, align=left] {\textbf{Extraction Workflow}};
    \end{tikzpicture}

    \vspace{0.1em}

    {\footnotesize
    \textbf{1. Financial Extraction (Prerequisite):}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item Extracts data from Snowflake (Cost Report, JOR, CRM).
        \item Updates \texttt{metadata\_extract\_status = 2} (Financial Complete).
        \item Schedule extraction cannot proceed until financial data is ready.
    \end{itemize}
    
    \vspace{0.1em}
    \textbf{2. User Request (Frontend):}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item User triggers extraction via UI.
        \item Creates submission record: \texttt{submission\_status = 0} (Pending).
    \end{itemize}

    \vspace{0.1em}
    \textbf{3. Schedule Extraction (Worker):}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item Picks up pending requests.
        \item Extracts P6 data (Activity, Calendar, Resource, Relationship).
        \item Calculates SHA256 hashes for each service.
        \item \textbf{Reconciliation:} Compares hashes with previous version. If changed, creates new version; otherwise, reuses version ID.
    \end{itemize}
    }

    {\footnotesize
    The system supports both \textbf{Weekly} and \textbf{Monthly} schedule publishers. Unlike the JSR (3 versions), the publishers and other schedules are built to compare \textbf{four} versions:
    
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item Current Period
        \item Previous Period
        \item Current Baseline
        \item Original Baseline
    \end{itemize}
    }

    % Schedule Lineage
    \begin{tikzpicture}
        \node[fill=black!10, rounded corners=3pt, inner sep=3pt, text width=\linewidth-6pt, align=left] {\textbf{3. Schedule Lineage}};
    \end{tikzpicture}
    
    \vspace{0.1em}
    
    {\footnotesize
    \textbf{Goal:} Massively re-use and track schedule changes.
    
    \vspace{0.15em}
    
    \textbf{Features:}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item Collects current\_schedule lineage at a project level.
        \item Enables time series analysis of WHEN a schedule was built (using \texttt{fiscal\_year\_month\_no}).
        \item Tracks IF there are changes between versions.
        \item Maintains complete lineage for each \texttt{project\_number} by version.
        \item Captures at least a monthly snapshot from the JSR.
    \end{itemize}
    }

    {\footnotesize
    \begin{itemize}[leftmargin=1em, itemsep=0em, topsep=0em]
        \item \textbf{Massive Performance:} Dagster runs parallel workstreams per project and per schedule; API serves from parquet.

        \item \textbf{Unified Queue:} JSR, Publishers, and Lineage share processing infrastructure.
        \item \textbf{Job Code System:} Expanded from Active SAP only to Estimates + Y-Past + Active projects.
        \item \textbf{Full Monitoring:} Observability built into Dagster (pipeline) and NextJS/React app (user-facing).

    \end{tikzpicture}

    \vspace{0.05em}

    {\footnotesize
    \textbf{1. Direct Parquet Read (No Iceberg Dependency)}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item JSR app reads from \textbf{aggregated parquet files} immediately after extraction.
        \item No waiting for Iceberg write to complete.
        \item Status 4 (Aggregated) is sufficient for JSR display.
    \end{itemize}

    \vspace{0.1em}
    \textbf{2. Batched Iceberg Writes}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item Iceberg writes batched in \textbf{15-minute intervals}.
        \item Prevents small file proliferation.
        \item Analytics/downstream consumers use Iceberg.
    \end{itemize}

    \vspace{0.1em}
    \textbf{3. Optimal CDC Strategy}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item Tables use \textbf{CDC (Change Data Capture)} to Iceberg when feasible.
        \item Falls back to \textbf{batch writes} if CDC would create too many small files.
        \item Balances real-time updates with storage efficiency.
    \end{itemize}
    }

    \end{tikzpicture}

    \vspace{0.05em}

    {\footnotesize
    \textbf{SHA256 Hash Comparison:}
    \begin{enumerate}[leftmargin=1.2em, itemsep=0em]
        \item Extract P6 data into memory.
        \item Calculate SHA256 hash for each service parquet.
        \item Compare with hashes in \texttt{p6\_schedule\_id\_versions}.
        \item If \textbf{ANY} hash differs $\rightarrow$ increment version.
        \item If \textbf{ALL} hashes match $\rightarrow$ reuse existing version.
    \end{enumerate}

    \vspace{0.1em}
    \textbf{Version Record Created:}
    \begin{itemize}[leftmargin=1em, itemsep=0em]
        \item \texttt{version\_number}: 1, 2, 3...
        \item \texttt{output\_dir\_url}: Full blob path
        \item \texttt{service\_*\_file\_hash}: SHA256 per service
        \item \texttt{service\_*\_row\_count}: Metadata
    \end{itemize}
    }
